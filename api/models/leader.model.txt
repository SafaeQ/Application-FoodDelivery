const mongoose = require('mongoose')
const RoleSchema = require('./user.model')

const Leader = mongoose.model(
    "Leader",
    new mongoose.Schema({
        name: {
            type: String,
          },
      
          password: {
            type: String,
          },
          email: {
            type: String,
            unique: true,
          },
          number: {
            type: Number,
          },
          role: RoleSchema,
          card: [
            quantity = {
              type: Number
            },
            foodId = {
              type: mongoose.Schema.Types.ObjectId,
              ref: 'Repast'
            }
          ]
        },
        {
          timestamps: true,
          versionKey: false,
        }
    )
)

module.exports = Leader
